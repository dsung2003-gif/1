function loop() { if (!isRunning) return; try { // 1. 비디오 상태 확인 if (video.paused || video.ended) return; const vw = video.videoWidth; const vh = video.videoHeight; if (vw === 0 || vh === 0) { requestAnimationFrame(loop); return; } // 2. Mat 객체 안전 생성 및 재사용 if (!src || src.cols !== vw || src.rows !== vh) { if (src) src.delete(); if (gray) gray.delete(); if (res) res.delete(); src = new cv.Mat(vh, vw, cv.CV_8UC4); gray = new cv.Mat(); res = new cv.Mat(); cap = new cv.VideoCapture(video); } // 3. 화면 읽기 시도 cap.read(src); if (src.empty()) { requestAnimationFrame(loop); return; } cv.cvtColor(src, gray, cv.COLOR_RGBA2GRAY); let bestMatch = { id: -1, val: 0 }; // 4. 템플릿 매칭 for (let t of templates) { // 중요: 템플릿 Mat이 유효한지, 화면보다 작은지 확인 if (!t.mat || t.mat.empty()) continue; if (t.mat.cols > gray.cols || t.mat.rows > gray.rows) continue; try { cv.matchTemplate(gray, t.mat, res, cv.TM_CCOEFF_NORMED); let m = cv.minMaxLoc(res); if (m.maxVal > bestMatch.val) { bestMatch = { id: t.id, val: m.maxVal }; } } catch (matchError) { // 특정 템플릿에서만 에러가 날 경우 건너뜀 continue; } } // 5. 결과 업데이트 if (bestMatch.val > 0.75) { status.innerText = `✅ 감지: ${bestMatch.id} (${Math.round(bestMatch.val * 100)}%)`; } else { status.innerText = `🔍 스캔 중... (최고: ${Math.round(bestMatch.val * 100)}%)`; } } catch (e) { // 어떤 오류인지 상세히 출력하여 디버깅을 돕습니다. console.warn("루프 내부 세부 오류:", e); } // 다음 프레임 예약 (setTimeout 대신 requestAnimationFrame 고려 가능하나 // 부하 조절을 위해선 setTimeout이 유리합니다) setTimeout(loop, 150); }

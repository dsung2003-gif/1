function loop() { if (!isRunning) return; try { const vw = video.videoWidth; const vh = video.videoHeight; // 1. 비디오 데이터가 아직 준비되지 않았다면 다음 프레임으로 토스 if (vw === 0 || vh === 0) { requestAnimationFrame(loop); return; } // 2. [핵심 수정] src Mat이 없거나 비디오 크기와 다르면 "반드시" 새로 생성 if (!src || src.cols !== vw || src.rows !== vh) { if (src) src.delete(); if (gray) gray.delete(); if (res) res.delete(); // 비디오 크기와 정확히 일치하는 Mat 생성 src = new cv.Mat(vh, vw, cv.CV_8UC4); gray = new cv.Mat(); res = new cv.Mat(); // VideoCapture 객체도 크기가 바뀔 때 새로 생성해주는 것이 안전합니다. cap = new cv.VideoCapture(video); console.log(`그릇 크기 재설정 완료: ${vw}x${vh}`); } // 3. 이제 크기가 일치하므로 안전하게 읽기 cap.read(src); // 4. 나머지 처리 (그레이스케일 변환 및 매칭) cv.cvtColor(src, gray, cv.COLOR_RGBA2GRAY); let bestMatch = { id: -1, val: 0 }; for (let t of templates) { if (t.mat.cols > gray.cols || t.mat.rows > gray.rows) continue; cv.matchTemplate(gray, t.mat, res, cv.TM_CCOEFF_NORMED); let m = cv.minMaxLoc(res); if (m.maxVal > bestMatch.val) { bestMatch = { id: t.id, val: m.maxVal }; } } // 결과 표시 if (bestMatch.val > 0.75) { status.innerText = `✅ 감지됨: ${bestMatch.id} (${Math.round(bestMatch.val * 100)}%)`; } else { status.innerText = `🔍 스캔 중... (최고 유사도: ${Math.round(bestMatch.val * 100)}%)`; } } catch (e) { // 오류가 발생하면 Mat을 초기화하여 다음 루프에서 재생성하도록 유도 console.warn("루프 내 에러 발생:", e); if (src) { src.delete(); src = null; } } setTimeout(loop, 150); }
